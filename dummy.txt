import java.io.*;
import java.net.*;
import java.util.*;

public class ChatServer_Data {

    static ArrayList<DataOutputStream> clientOutputs = new ArrayList<>();
    static int clientNumber = 0;

    public static void main(String[] args) throws Exception {

        ServerSocket ss = new ServerSocket(5000);
        System.out.println("Server started...");

        while (true) {

            Socket client = ss.accept();
            clientNumber++;

            String clientName = "Client" + clientNumber;
            System.out.println(clientName + " connected.");

            DataOutputStream out = new DataOutputStream(client.getOutputStream());
            DataInputStream in = new DataInputStream(client.getInputStream());

            clientOutputs.add(out);

            // Send welcome message
            out.writeUTF("Welcome " + clientName);

            // Thread to listen for messages
            new Thread(() -> {
                try {
                    while (true) {
                        String msg = in.readUTF();
                        System.out.println(clientName + ": " + msg);

                        // Broadcast to all other clients
                        for (DataOutputStream dos : clientOutputs) {
                            if (dos != out) {
                                dos.writeUTF(clientName + ": " + msg);
                            }
                        }
                    }
                } catch (Exception e) {
                    System.out.println(clientName + " disconnected.");
                }
            }).start();
        }
    }
}


import java.io.*;
import java.net.*;
import java.util.*;

public class ChatClient_Data {

    public static void main(String[] args) throws Exception {

        Socket socket = new Socket("localhost", 5000);
        System.out.println("Connected to server.");

        DataOutputStream out = new DataOutputStream(socket.getOutputStream());
        DataInputStream in = new DataInputStream(socket.getInputStream());

        Scanner sc = new Scanner(System.in);

        // Thread to listen for server messages
        new Thread(() -> {
            try {
                while (true) {
                    String msg = in.readUTF();
                    System.out.println(msg);
                }
            } catch (Exception e) {
                System.out.println("Server disconnected.");
            }
        }).start();

        // Main send loop
        while (true) {
            String msg = sc.nextLine();
            out.writeUTF(msg);
        }
    }
}
